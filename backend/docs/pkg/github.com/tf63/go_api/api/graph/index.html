type ComplexityRoot struct {
    Expense struct {
        CreatedAt func(childComplexity int) int
        ID        func(childComplexity int) int
        Price     func(childComplexity int) int
        Title     func(childComplexity int) int
        UpdatedAt func(childComplexity int) int
        UserID    func(childComplexity int) int
    }

    Mutation struct {
        CreateExpense func(childComplexity int, input NewExpense) int
        CreateUser    func(childComplexity int, input NewUser) int
        DeleteExpense func(childComplexity int, input FindUser, expenseID uint) int
        DeleteUser    func(childComplexity int, userID uint) int
        UpdateExpense func(childComplexity int, input NewExpense, expenseID uint) int
        UpdateUser    func(childComplexity int, input NewUser, userID uint) int
    }

    Query struct {
        Expense  func(childComplexity int, input FindUser, expenseID uint) int
        Expenses func(childComplexity int, input FindUser) int
        User     func(childComplexity int, input FindUser) int
        Users    func(childComplexity int) int
    }

    User struct {
        CreatedAt func(childComplexity int) int
        ID        func(childComplexity int) int
        Name      func(childComplexity int) int
        UpdatedAt func(childComplexity int) int
    }
}
